import requests

def get_binance_futures_data(symbol="BTCUSDT"):
    base_url = "https://fapi.binance.com"
    long_short_url = f"{base_url}/futures/data/globalLongShortAccountRatio?symbol={symbol}&period=30m&limit=1"
    funding_url = f"{base_url}/fapi/v1/fundingRate?symbol={symbol}&limit=1"
    oi_url = f"{base_url}/fapi/v1/openInterest?symbol={symbol}"

    long_short = requests.get(long_short_url).json()[0]
    funding = requests.get(funding_url).json()[0]
    oi = requests.get(oi_url).json()

    return {
        "long_short_ratio": float(long_short['longShortRatio']),
        "long_account": float(long_short['longAccount']),
        "short_account": float(long_short['shortAccount']),
        "funding_rate": float(funding['fundingRate']),
        "open_interest": float(oi['openInterest'])
    }
