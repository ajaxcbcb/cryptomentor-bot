import os

def is_admin(user_id) -> bool:
    if user_id is None:
        return False
    uid = str(user_id)
    a1 = (os.getenv("ADMIN_USER_ID") or "").strip()
    a2 = (os.getenv("ADMIN2_USER_ID") or "").strip()
    return uid == a1 or uid == a2
