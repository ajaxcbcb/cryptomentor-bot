1) Util concurrent yang rapi

Buat app/utils/async_tools.py:

import asyncio
from typing import Iterable, Awaitable, List, Any

async def gather_safe(tasks: Iterable[Awaitable[Any]]) -> List[Any]:
    """
    Jalankan semua task async secara concurrent dan kembalikan hasil dalam urutan input.
    Jika ada exception pada salah satu task, hasil indeks tersebut akan berupa exception.
    """
    return await asyncio.gather(*tasks, return_exceptions=True)

Edit app/services/analysis.py:

from typing import Dict, Any, List
import asyncio
import pandas as pd
from app.utils.async_tools import gather_safe
from app.providers.coinapi import get_ohlcv, get_price_spot
from app.services.indicators import to_df, ema, rsi, macd, atr

def _trend_label(df: pd.DataFrame) -> str:
    e50 = ema(df["close"], 50)
    e200 = ema(df["close"], 200)
    if e50.iloc[-1] > e200.iloc[-1]: return "up"
    if e50.iloc[-1] < e200.iloc[-1]: return "down"
    return "sideways"

def _single_entry_point(df: pd.DataFrame) -> float:
    last_close = float(df["close"].iloc[-1])
    e50 = ema(df["close"], 50).iloc[-1]
    a14 = atr(df, 14).iloc[-1]
    trend = _trend_label(df)
    if trend == "up":
        return float(min(e50, last_close - 0.5 * a14))
    elif trend == "down":
        return float(max(e50, last_close + 0.5 * a14))
    else:
        return float(last_close - 0.2 * a14)

async def analyze_coin_futures(coin: str) -> Dict[str, Any]:
    """
    Analisis khusus futures (perp) dan kembalikan dict siap pakai, bukan coroutine.
    """
    ohlcv = await get_ohlcv(coin, period="5MIN", limit=300, market="perp")
    df = to_df(ohlcv).dropna()
    trend = _trend_label(df)
    r = rsi(df["close"])
    m, s, h = macd(df["close"])
    a = atr(df, 14)

    res: Dict[str, Any] = {
        "coin": coin.upper(),
        "price": float(df["close"].iloc[-1]),
        "trend": trend,
        "rsi": float(r.iloc[-1]),
        "macd_hist": float(h.iloc[-1]),
        "atr": float(a.iloc[-1]),
        "entry": _single_entry_point(df),  # satu titik
    }
    # Sinyal sederhana
    res["ok"] = (res["trend"] == "up") and (res["macd_hist"] > 0) and (40 < res["rsi"] < 70)
    return res

async def futures_signals(coins: List[str]) -> List[Dict[str, Any]]:
    """
    Scan koin futures secara concurrent & aman.
    HASIL = list of dict (bukan coroutine), sehingga formatter .get() aman.
    """
    coins_up = [c.upper() for c in coins]
    tasks = [analyze_coin_futures(c) for c in coins_up]  # <— TASKS (await nanti)
    results = await gather_safe(tasks)                   # <— AWAIT SEMUA

    out: List[Dict[str, Any]] = []
    for c, r in zip(coins_up, results):
        if isinstance(r, Exception):
            out.append({"coin": c, "error": str(r)})
        else:
            # r sudah dict siap .get()
            out.append(r)
    return out


Catatan:

Tidak ada .get pada coroutine karena semua task sudah await via gather_safe.

Jika satu koin error, hanya item itu yang berisi {"error": ...}.

from app.services.analysis import futures_signals

async def cmd_futures_signals(update, context):
    coins = [c.upper() for c in (context.args or ["BTC","ETH","BNB"])]
    try:
        lst = await futures_signals(coins)  # <— PENTING: await
        # Formatter lama
        try:
            import importlib
            F = importlib.import_module("app.formatters.texts")
        except Exception:
            F = None

        if F and hasattr(F, "format_futures_signals"):
            text = F.format_futures_signals(lst)  # lst = list[dict], aman .get()
        else:
            # Fallback ringkas
            lines = ["🛰️ Futures Signals"]
            for it in lst:
                if "error" in it:
                    lines.append(f"- {it['coin']}: ❌ error")
                else:
                    lines.append(
                        f"- {it['coin']}: {'✅' if it.get('ok') else '❌'} | "
                        f"entry {it.get('entry'):.4f} | "
                        f"rsi {it.get('rsi'):.1f} | macd {it.get('macd_hist'):.4f} | "
                        f"{it.get('trend')}"
                    )
            text = "\n".join(lines)
        await update.effective_message.reply_text(text)
    except Exception as e:
        await update.effective_message.reply_text(
            "❌ **Terjadi kesalahan mengambil data untuk FUTURES_SIGNALS**\n\n"
            "💡 **Coba alternatif**:\n"
            "• `/price btc` - Cek harga basic\n"
            "• Tunggu beberapa menit dan coba lagi\n"
            "• Hubungi admin jika masalah berlanjut\n\n"
            "🔄 **Error context**: scan process\n",
            parse_mode="Markdown"
        )
